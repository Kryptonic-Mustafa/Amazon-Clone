"use client";

import { useEffect, useState, useRef } from 'react';
import Link from 'next/link';
import Image from 'next/image';
import { ArrowRight, ChevronRight, ChevronLeft, Phone, Mail, MapPin } from 'lucide-react';
import Header from '@/components/shop/Header';
import Footer from '@/components/shop/Footer';
import GoogleTranslate from '@/components/shop/GoogleTranslate';
import { LanguageProvider } from '@/context/LanguageContext';
import { CurrencyProvider } from '@/context/CurrencyContext';
import { SettingsProvider } from '@/context/SettingsContext';

function HomeContent() {
  const [featuredProducts, setFeaturedProducts] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);
  const carouselRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    const fetchProducts = async () => {
      try {
        const res = await fetch('/api/shop/products');
        const data = await res.json();
        if (Array.isArray(data)) setFeaturedProducts(data);
      } catch (e) {
        console.error("Error loading products", e);
      } finally {
        setLoading(false);
      }
    };
    fetchProducts();
  }, []);

  const scroll = (direction: 'left' | 'right') => {
    if (carouselRef.current) {
      const { scrollLeft, clientWidth } = carouselRef.current;
      const scrollTo = direction === 'left' ? scrollLeft - clientWidth : scrollLeft + clientWidth;
      carouselRef.current.scrollTo({ left: scrollTo, behavior: 'smooth' });
    }
  };

  return (
    <div className="bg-white min-h-screen flex flex-col text-black">
      <GoogleTranslate />
      <Header />
      
      {/* HERO SECTION - STATIC FOR TRANSLATION */}
      <section className="relative bg-slate-900 text-white py-24 px-6 min-h-[600px] flex items-center">
        <div className="absolute inset-0 z-0 opacity-40">
             <Image src="https://images.unsplash.com/photo-1441986300917-64674bd600d8" alt="Store Background" fill className="object-cover" priority />
        </div>
        <div className="container mx-auto relative z-10 text-center md:text-left">
          <h1 className="text-4xl md:text-7xl font-extrabold mb-6 leading-tight">
            Shop the World's <br/> <span className="text-yellow-400">Best Deals</span>
          </h1>
          <p className="text-lg md:text-2xl mb-10 max-w-2xl text-slate-200">
            Find everything from high-end electronics to daily essentials with fast shipping and secure payments.
          </p>
          <Link href="/shop" className="bg-yellow-400 hover:bg-yellow-500 text-black px-10 py-4 rounded-lg font-bold text-lg transition-all inline-flex items-center gap-2 shadow-xl hover:scale-105 active:scale-95">
            Start Shopping <ArrowRight size={20} />
          </Link>
        </div>
      </section>

      {/* FEATURED PRODUCTS CAROUSEL (RESTORED) */}
      <section className="py-20 px-4 bg-white overflow-hidden">
        <div className="container mx-auto">
          <div className="flex items-center justify-between mb-10 px-2">
            <div>
                <h2 className="text-3xl font-bold text-black">Featured Products</h2>
                <div className="h-1 w-20 bg-yellow-400 mt-2"></div>
            </div>
            <div className="flex gap-3">
              <button onClick={() => scroll('left')} className="p-3 rounded-full border-2 border-gray-100 hover:bg-gray-100 transition-colors"><ChevronLeft/></button>
              <button onClick={() => scroll('right')} className="p-3 rounded-full border-2 border-gray-100 hover:bg-gray-100 transition-colors"><ChevronRight/></button>
            </div>
          </div>

          {loading ? (
            <div className="flex gap-6 overflow-hidden">
                {[1,2,3,4].map(i => <div key={i} className="min-w-[280px] h-80 bg-gray-100 animate-pulse rounded-xl"></div>)}
            </div>
          ) : (
            <div ref={carouselRef} className="flex overflow-x-auto gap-6 pb-6 scrollbar-hide no-scrollbar" style={{ scrollbarWidth: 'none', msOverflowStyle: 'none' }}>
                {featuredProducts.map((product) => (
                <div key={product.id} className="min-w-[280px] border border-gray-100 rounded-2xl p-5 hover:shadow-2xl transition-all bg-white group">
                    <div className="h-48 relative mb-6 bg-gray-50 rounded-xl overflow-hidden">
                    {product.image_urls && (
                        <Image 
                            src={product.image_urls.split(',')[0]} 
                            alt={product.name} 
                            fill 
                            className="object-contain group-hover:scale-110 transition-transform duration-300" 
                        />
                    )}
                    </div>
                    <h3 className="font-bold text-black text-lg truncate mb-1">{product.name}</h3>
                    <p className="text-gray-500 text-sm mb-3 line-clamp-1">{product.brand || 'Premium Brand'}</p>
                    <div className="flex items-center justify-between">
                        <p className="text-blue-600 font-extrabold text-xl">${product.price}</p>
                        <button className="bg-slate-900 text-white p-2 rounded-lg hover:bg-blue-600 transition-colors">
                            <ArrowRight size={18}/>
                        </button>
                    </div>
                </div>
                ))}
            </div>
          )}
        </div>
      </section>

      {/* ABOUT US - STATIC */}
      <section className="py-24 px-4 bg-slate-50">
        <div className="container mx-auto max-w-6xl grid md:grid-cols-2 gap-16 items-center">
          <div className="relative h-[450px] rounded-3xl overflow-hidden shadow-2xl border-8 border-white">
             <Image src="https://images.unsplash.com/photo-1522202176988-66273c2fd55f" alt="About Our Team" fill className="object-cover" />
          </div>
          <div className="text-black">
            <h3 className="text-blue-600 font-bold uppercase text-sm tracking-[0.2em] mb-4">Quality & Trust</h3>
            <h2 className="text-4xl md:text-5xl font-extrabold mb-8 leading-tight">Your Global Shopping <br/> Partner.</h2>
            <p className="text-lg text-gray-600 mb-8 leading-relaxed">
              At AmazonClone, we believe that quality shouldn't come with a high price tag. We bridge the gap between world-class brands and your doorstep, ensuring every purchase is a step toward excellence.
            </p>
            <div className="grid grid-cols-2 gap-6">
                <div className="p-4 bg-white rounded-xl shadow-sm">
                    <h4 className="font-bold text-blue-600 text-2xl mb-1">24/7</h4>
                    <p className="text-sm text-gray-500 font-medium">Customer Support</p>
                </div>
                <div className="p-4 bg-white rounded-xl shadow-sm">
                    <h4 className="font-bold text-blue-600 text-2xl mb-1">100%</h4>
                    <p className="text-sm text-gray-500 font-medium">Secure Payments</p>
                </div>
            </div>
          </div>
        </div>
      </section>

      <Footer />
    </div>
  );
}

export default function HomePage() {
  return (
    <SettingsProvider>
      <LanguageProvider>
        <CurrencyProvider>
          <HomeContent />
        </CurrencyProvider>
      </LanguageProvider>
    </SettingsProvider>
  );
}
